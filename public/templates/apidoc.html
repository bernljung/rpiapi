$url = $host.'/api/v1/speak';

$fields = array(
            'speech' => "I'm going to make him an offer he can't refuse.",
            'lang' => 'en'
        );

$fields_string = "";
foreach ($fields as $key => $value) {
        $fields_string .= $key . '=' . $value . '&';
}
rtrim($fields_string, '&');

$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $url);
curl_setopt($ch, CURLOPT_POST, count($fields));
curl_setopt($ch, CURLOPT_POSTFIELDS, $fields_string);
$result = curl_exec($ch);
curl_close($ch);
